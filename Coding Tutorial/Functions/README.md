# Function:
A Function is Block of Reusable Code.

```cpp
// functions = A Block Of Reusable Code

#include <iostream>
using namespace std;

void mithundama () {
    cout << "He is Great Man" << "\n";
    cout << "He is Greatest of all time" << "\n";
    cout << "He is a real legend" << "\n";
    cout << "He is Great Man" << "\n";
    cout << "He is Great Man" << "\n\n";

}

int main() {
    mithundama();
    mithundama();
    mithundama();
    mithundama();
    return 0;
}
```
